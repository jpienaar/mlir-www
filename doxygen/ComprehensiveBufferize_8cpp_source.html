9 // Perform inplace bufferization within function boundaries.
10 // This is a specialized pass that supports inplace analysis for a fixed subset
11 // of ops that have well-defined inplace semantics.
12 // This pass caters to high-performance codegen where buffer reuse is deemed
13 // critical: the pass should fail if the bufferized form of the function needs
14 // to return any buffer.
15 // Generic control-flow and branching are unsupported.
16 // Composability with extensible set of ops is not a first-class concern.
17 //
18 // Bufferization occurs by:
19 // a. performing an inPlace analysis `inPlaceAnalysisFuncOpBody`
20 // which marks each operation within the function with the
21 // `kInPlaceResultsAttrName` attribute.
22 // b. traversing each operation in the function and rewriting it in
23 // buffer form and keeping a BlockAndValueMapping mapping of the
24 // rewrites. New allocations are introduced during this step.
25 // TODO: Allocation + depending op hoisting to outermost enclosing
26 // sequential scope.
27 // c. at the end of this bufferization, 3 cases may occur:
28 // i. inplaceable function arguments may be reused in place after the
29 // function itself has been bufferized. This is encoded by IR resembling:
30 //
31 // ```
32 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
33 // func @foo(%A: tensor<?xf32> {linalg.inplaceable = true})
34 // -> tensor<?xf32> {
35 // %0 = memref.buffer_cast %A : memref<?xf32, #map>
36 // // ... uses of %0
37 // %res = memref.tensor_load %0 : memref<?xf32, #map>
38 // return %res : tensor<?xf32>
39 // }
40 // ```
41 //
42 // this is the cue for the bufferization of the function foo (and calls
43 // to it) may bufferize to `func @foo(%A: memref<?xf32, some_layout>)`.
44 // To fully achieve bufferization, an additional analysis is needed to
45 // determine whether function argument/operand pairs bufferize to a
46 // single inplace buffer argument (i.e. functions may return tensors in
47 // arbitrary order that may not match argument numbers).
48 //
49 // ii. results that don't map to an inplaceable function argument are
50 // generally allocated. Since memref semantics wrt ownership of the
51 // underlying memory region are not well-defined, comprehensive
52 // bufferization chooses to perform allocations in a scoped fashion:
53 // returning memrefs is always considered illegal.
54 // Such scenarios are encoded by IR resembling:
55 //
56 // ```
57 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
58 // func @foo(%A: tensor<?xf32> {linalg.inplaceable = true})
59 // -> tensor<?xf32> {
60 // %0 = memref.buffer_cast %A : memref<?xf32, #map>
61 // %1 = memref.dim %0, %c0 : memref<?xf32, #map>
62 // %2 = memref.alloc(%1) : memref<?xf32>
63 // %3 = memref.cast %2 : memref<?xf32> to memref<?xf32, #map>
64 // // ... uses of %3
65 // memref.dealloc %2 : memref<?xf32, #map>
66 // %res = memref.tensor_load %3 : memref<?xf32, #map>
67 // return %res : tensor<?xf32>
68 // }
69 // ```
70 //
71 // this is the cue for the bufferization of the function foo (and calls
72 // to it) that it must bufferize to `func @foo(%A: memref<?xf32,
73 // some_layout>,
74 // %B: memref<?xf32, some_layout>)` (i.e. make a cloned
75 // allocation of the result tensor)
76 // To fully achieve bufferization, the alloc/dealloc pair must be lifted
77 // out of the function at each call site.
78 //
79 // iii. as an optimization over ii., it may be possible to reuse an argument
80 // and only want to return a slice.
81 // This may forego allocation by letting *all* callers decide whether to
82 // pass a new *aliasing* memref function argument (i.e. a subview).
83 // Without loss of generality, callers may agree to allocate a new buffer
84 // to avoid this aliasing. Such scenarios are encoded by IR resembling:
85 //
86 // ```
87 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
88 // func @foo(%arg0: tensor<?xf32> {linalg.inplaceable = true})
89 // -> tensor<4xf32> {
90 // %0 = memref.buffer_cast %arg0 : memref<?xf32, #map>
91 // %1 = memref.subview %0[0] [4] [1] : memref<?xf32, #map> to
92 // memref<4xf32, #map>
93 // // ... inplace computes into %1
94 // %3 = memref.tensor_load %1 : memref<4xf32, #map>
95 // return %3 : tensor<4xf32>
96 // }
97 // ```
98 //
99 // Note: In the future, it may be worthwhile to design special bufferization
100 // ops to encode the desired semantics at function boundaries for i., ii. and
101 // iii.
102 //
103 // Lastly, note that layout map chosen to bufferize is the most dynamic
104 // canonical strided layout of the proper rank. This ensures compatibility with
105 // expected layouts after transformations. Combinations of memref.cast +
106 // canonicalization are responsible for clean ups.

267 /// Mark whether OpResult can actually be bufferized inplace.
268 /// If `inPlace` is `InPlaceSpec::True`, the use-def chain analysis has
269 /// guaranteed that no subsequent write would occur to the bufferized
270 /// tensor value (i.e. the result can be bufferized inPlace).
271 static void setInPlaceOpResult(OpResult opResult,
272  InPlaceSpec inPlace = InPlaceSpec::True) {
273  if (!opResult)
274  return;
275 
276  Operation *op = opResult.getOwner();
277  auto attr =
278  op->getAttr(kInPlaceResultsAttrName).dyn_cast_or_null<ArrayAttr>();
279  SmallVector<StringRef> inPlaceVector =
280  attr ? SmallVector<StringRef>(
281  llvm::to_vector<4>(attr.getAsValueRange<StringAttr>()))
282  : SmallVector<StringRef>(op->getNumResults(),
283  stringify(InPlaceSpec::None));
284  LDBG("->set inPlace=" << stringify(inPlace) << " <- #"
285  << opResult.getResultNumber() << ": "
286  << printOperationInfo(op) << "\n");
287  inPlaceVector[opResult.getResultNumber()] = stringify(inPlace);
288  op->setAttr(kInPlaceResultsAttrName,
289  OpBuilder(op).getStrArrayAttr(inPlaceVector));
290 }

292 /// Get the InPlaceSpec attribute entry `kInPlaceResultsAttrName` for
293 /// `opResult`. If the result is `InPlaceSpec::True`, the use-def chain analysis
294 /// has guaranteed that no subsequent read of the tensor value occurs and the
295 /// result can be buferized inPlace.
296 /// If no InPlaceSpec attribute has been set for `opResult`, return
297 /// InPlaceSpec::None.
298 static InPlaceSpec getInPlace(OpResult opResult) {
299  if (!opResult)
300  return InPlaceSpec::None;
301 
302  Operation *op = opResult.getOwner();
303  auto attr =
304  op->getAttr(kInPlaceResultsAttrName).dyn_cast_or_null<ArrayAttr>();
305  if (!attr)
306  return InPlaceSpec::None;
307 
308  // Must return a proper value.
309  return *symbolize(*(attr.getAsValueRange<StringAttr>().begin() +
310  opResult.getResultNumber()));
311 }

313 /// Get inPlace information for `bbArg`.
314 /// FuncOp allow argument attributes, we use those to encode the information.
315 /// BlockArgument of other ops delegate to their owner's parent op.
316 static InPlaceSpec getInPlace(BlockArgument bbArg) {
317  if (auto funcOp = dyn_cast<FuncOp>(bbArg.getOwner()->getParentOp())) {
318  BoolAttr inplaceAttr = funcOp.getArgAttrOfType<BoolAttr>(
319  bbArg.getArgNumber(), LinalgDialect::kInplaceableAttrName);
320  if (!inplaceAttr)
321  return InPlaceSpec::None;
322  return inplaceAttr.getValue() ? InPlaceSpec::True : InPlaceSpec::False;
323  }
324  // Interestingly, scf::ForOp's and TiledLoop's bbArg can **always** be viewed
325  // inplace from the perspective of ops nested under:
326  // 1. Either the matching iter operand is not bufferized inplace and an
327  // alloc + optional copy makes the bbArg itself inplaceable.
328  // 2. Or the matching iter operand is bufferized inplace and bbArg just
329  // bufferizes to that too.
330  if (isa<scf::ForOp, TiledLoopOp>(bbArg.getOwner()->getParentOp()))
331  return InPlaceSpec::True;
332  // Unknown cases.
333  return InPlaceSpec::None;
334 }

568  SmallVector<OpOperand *> r;
569  // Unknown ops are handled conservatively and never bufferize in-place.
570  if (!hasKnownBufferizationAliasingBehavior(result.getDefiningOp()))
571  return SmallVector<OpOperand *>();
572  TypeSwitch<Operation *>(result.getDefiningOp())
573  .Case([&](tensor::CastOp op) { r.push_back(&op->getOpOperand(0)); })
574  .Case([&](ExtractSliceOp op) { r.push_back(&op->getOpOperand(0)); })
575  .Case([&](scf::IfOp op) { populateAliasingOpOperands(op, result, r); })
576  // In the case of scf::ForOp, this currently assumes the iter_args / yield
577  // are 1-1. This may fail and is verified at the end.
578  // TODO: update this.
579  .Case([&](scf::ForOp op) {
580  r.push_back(&op.getIterOpOperands()[result.getResultNumber()]);

582  .Case([&](InsertSliceOp op) { r.push_back(&op->getOpOperand(1)); })
583  .Case([&](LinalgOp op) {
584  r.push_back(op.getOutputTensorOperands()[result.getResultNumber()]);
585  })
586  .Case([&](TiledLoopOp op) {
587  // TODO: TiledLoopOp helper method to avoid leaking impl details.
588  r.push_back(&op->getOpOperand(op.getNumControlOperands() +
589  op.getNumInputs() +
590  result.getResultNumber()));

592  .Case([&](vector::TransferWriteOp op) {
593  r.push_back(&op->getOpOperand(1));

595  .Case<arith::ConstantOp, ConstantOp, CallOpInterface, InitTensorOp>(
596  [&](auto op) {})
597  .Default([&](Operation *op) {
598  op->dump();
599  llvm_unreachable("unexpected defining op");
600  });

602 }

630 /// Return true if the given value is read by an op that bufferizes to a memory
631 /// read. Also takes into account ops that create an alias but do not read by
632 /// themselves (e.g., ExtractSliceOp).
633 static bool isValueRead(Value value) {
634  SmallVector<OpOperand *> workingSet;
635  for (OpOperand &use : value.getUses())
636  workingSet.push_back(&use);
637 
638  while (!workingSet.empty()) {
639  OpOperand *uMaybeReading = workingSet.pop_back_val();
640  // Skip over all ExtractSliceOps. These do not read by themselves but just
641  // add a new alias.
642  if (auto extractSliceOp =
643  dyn_cast<ExtractSliceOp>(uMaybeReading->getOwner()))
644  for (OpOperand &use : extractSliceOp.result().getUses())
645  workingSet.push_back(&use);
646  if (bufferizesToMemoryRead(*uMaybeReading))
647  return true;
648  }

653 /// Return true if `opOperand` bufferizes to a memory read.
654 static bool bufferizesToMemoryRead(OpOperand &opOperand) {
655  // Unknown op that returns a tensor. The inplace analysis does not support
656  // it. Conservatively return true.
657  if (!hasKnownBufferizationAliasingBehavior(opOperand.getOwner()))
658  return true;
659  // ExtractSliceOp alone doesn't bufferize to a memory read, one of its uses
660  // may.
661  if (isa<ExtractSliceOp>(opOperand.getOwner()))
662  return false;
663  // scf::ForOp alone doesn't bufferize to a memory read, one of the uses of its
664  // matching bbArg may.
665  if (auto forOp = dyn_cast<scf::ForOp>(opOperand.getOwner()))
666  return isValueRead(forOp.getRegionIterArgForOpOperand(opOperand));
667  // TiledLoop alone doesn't bufferize to a memory read, one of the uses of its
668  // matching bbArg may.
669  if (auto tiledLoopOp = dyn_cast<TiledLoopOp>(opOperand.getOwner()))
670  return isValueRead(tiledLoopOp.getTiedBlockArgument(opOperand));
671  // CallOpInterface alone doesn't bufferize to a memory read, one of the uses
672  // of the matching bbArg may. It is the responsibility of the caller to
673  // inspect bbArgs. In the absence of a BufferizationAliasInfo, we need to be
674  // conservative.
675  if (auto callOp = dyn_cast<CallOpInterface>(opOperand.getOwner()))
676  return true;
677  if (auto linalgOp = dyn_cast<LinalgOp>(opOperand.getOwner()))
678  return linalgOp.isInputTensor(&opOperand) ||
679  linalgOp.isInitTensor(&opOperand);
680  // All other cases are considered to bufferize to memory reads.
681  // In particular, terminators are often the last use and need to be considered
682  // as reads to return the proper value and avoid WAW clobbers.
683  return true;
684 }

687 static bool bufferizesToMemoryWrite(OpOperand &opOperand) {
688  // These terminators are not writes.
689  if (isa<ReturnOp, linalg::YieldOp, scf::YieldOp>(opOperand.getOwner()))
690  return false;
691  // ExtractSliceOp alone doesn't bufferize to a memory write, one of its uses
692  // may.
693  if (isa<ExtractSliceOp>(opOperand.getOwner()))
694  return false;
695  // CallOpInterface alone doesn't bufferize to a memory write, one of the uses
696  // of the matching bbArg may. It is the responsibility of the caller to
697  // inspect bbArgs. In the absence of a BufferizationAliasInfo, we need to be
698  // conservative.
699  if (auto callOp = dyn_cast<CallOpInterface>(opOperand.getOwner()))
700  return true;
701  // Unknown op that returns a tensor. The inplace analysis does not support
702  // it. Conservatively return true.
703  if (!hasKnownBufferizationAliasingBehavior(opOperand.getOwner()))
704  return true;
705  OpResult opResult = getAliasingOpResult(opOperand);
706  // Only supported op with a matching result for opOperand bufferize to a
707  // write. E.g., ReturnOp does not bufferize to a write.

861 /// Starting from `value`, follow the use-def chain in reverse, always selecting
862 /// the aliasing OpOperands. Find and return Values for which `condition`
863 /// evaluates to true. OpOperands of such matching Values are not traversed any
864 /// further.
865 ///
866 /// When reaching the end of a chain (BlockArgument or Value without aliasing
867 /// OpOperands), also return the last Value of that chain.
868 ///
869 /// Example:
870 ///
871 /// 8
872 /// |
873 /// 6* 7* +-----+----+
874 /// | | | |
875 /// 2* 3 4* 5
876 /// | | | |
877 /// +----------+----------+----------+
878 /// |
879 /// 1
880 ///
881 /// In the above example, Values with a star satisfy the condition. When
882 /// starting the traversal from Value 1, the resulting SetVector is:
883 /// { 2, 7, 8, 5 }
884 static llvm::SetVector<Value>
885 findValueInReverseUseDefChain(Value value,
886  std::function<bool(Value)> condition) {
887  llvm::SetVector<Value> result, workingSet;
888  workingSet.insert(value);
889 
890  while (!workingSet.empty()) {
891  Value value = workingSet.pop_back_val();
892  if (condition(value) || value.isa<BlockArgument>()) {
893  result.insert(value);
894  continue;
895  }
896 
897  OpResult opResult = value.cast<OpResult>();
898  SmallVector<OpOperand *> opOperands = getAliasingOpOperand(opResult);
899  if (opOperands.empty()) {
900  result.insert(value);
901  continue;
902  }
903 
904  for (OpOperand *o : opOperands)
905  workingSet.insert(o->get());
906  }
907 
908  return result;
909 }

911 /// Find the Value of the last preceding write of a given Value.
912 ///
913 /// Note: Unknown ops are handled conservatively and assumed to be writes.
914 /// Furthermore, BlockArguments are also assumed to be writes. There is no
915 /// analysis across block boundaries.
916 ///
917 /// Note: To simplify the analysis, scf.if ops are considered writes. Treating
918 /// a non-writing op as a writing op may introduce unnecessary out-of-place
919 /// bufferizations, but is always safe from a correctness point of view.
920 static Value findLastPrecedingWrite(Value value) {
921  SetVector<Value> result =
922  findValueInReverseUseDefChain(value, [](Value value) {
923  Operation *op = value.getDefiningOp();
924  if (!op)
925  return true;
926  if (!hasKnownBufferizationAliasingBehavior(op))
927  return true;
928  if (isa<scf::IfOp>(op))
929  return true;
930 
931  SmallVector<OpOperand *> opOperands =
932  getAliasingOpOperand(value.cast<OpResult>());
933  assert(opOperands.size() <= 1 &&
934  "op with multiple aliasing OpOperands not expected");
935 
936  if (opOperands.empty())
937  return true;
938 
939  return bufferizesToMemoryWrite(*opOperands.front());
940  });
941  assert(result.size() == 1 && "expected exactly one result");
942  return result.front();
943 }

975 /// Given sets of uses and writes, return true if there is a RaW conflict under
976 /// the assumption that all given reads/writes alias the same buffer and that
977 /// all given writes bufferize inplace.
978 ///
979 /// A conflict is: According to SSA use-def chains, a read R is supposed to read
980 /// the result of a write W1. But because of bufferization decisions, R actually
981 /// reads another write W2.
982 bool BufferizationAliasInfo::hasReadAfterWriteInterference(
983  const DenseSet<OpOperand *> &usesRead,
984  const DenseSet<OpOperand *> &usesWrite,
985  const DominanceInfo &domInfo) const {
986 
987  for (OpOperand *uRead : usesRead) {
988  Operation *readingOp = uRead->getOwner();
989 
990  // Find most recent write of uRead by following the SSA use-def chain. E.g.:
991  //
992  // %0 = "writing_op"(%t) : tensor<?x32> -> tensor<?xf32>
993  // %1 = "aliasing_op"(%0) : tensor<?x32> -> tensor<?xf32>
994  // %2 = "reading_op"(%1) : : tensor<?x32> -> not_a_tensor_type
995  //
996  // In the above example, if uRead is the OpOperand of reading_op, lastWrite
997  // is %0. Note that operations that create an alias but do not write (such
998  // as ExtractSliceOp) are skipped.
999  Value lastWrite = findLastPrecedingWrite(uRead->get());
1000 
1001  // Look for conflicting memory writes. Potential conflicts are writes to an
1002  // alias that have been decided to bufferize inplace.
1003  for (OpOperand *uConflictingWrite : usesWrite) {
1004  // Throughout this loop, check for multiple requirements that have to be
1005  // met for uConflictingWrite to be an actual conflict.
1006  Operation *conflictingWritingOp = uConflictingWrite->getOwner();
1007 
1008  // Print some debug info.
1009  LDBG("Found potential conflict:\n");
1010  LDBG("READ = #" << uRead->getOperandNumber() << " of "
1011  << printOperationInfo(readingOp) << "\n");
1012  LDBG("CONFLICTING WRITE = #"
1013  << uConflictingWrite->getOperandNumber() << " of "
1014  << printOperationInfo(conflictingWritingOp) << "\n");
1015 
1016  // No conflict if the readingOp dominates conflictingWritingOp, i.e., the
1017  // write is not visible when reading.
1018  if (happensBefore(readingOp, conflictingWritingOp, domInfo))
1019  continue;
1020 
1021  // No conflict if the reading use equals the use of the conflicting write.
1022  // A use cannot conflict with itself. Note: Just being the same op is not
1023  // enough. It has to be the same use.
1024  if (uConflictingWrite == uRead)
1025  continue;
1026 
1027  if (scf::insideMutuallyExclusiveBranches(readingOp, conflictingWritingOp))
1028  continue;
1029 
1030  LDBG("WRITE = #" << printValueInfo(lastWrite) << "\n");
1031 
1032  // No conflict if the conflicting write happens before the last
1033  // write.
1034  if (Operation *writingOp = lastWrite.getDefiningOp()) {
1035  if (happensBefore(conflictingWritingOp, writingOp, domInfo))
1036  // conflictingWritingOp happens before writingOp. No conflict.
1037  continue;
1038  // No conflict if conflictingWritingOp is contained in writingOp.
1039  if (writingOp->isProperAncestor(conflictingWritingOp))
1040  continue;
1041  } else {
1042  auto bbArg = lastWrite.cast<BlockArgument>();
1043  Block *block = bbArg.getOwner();
1044  if (!block->findAncestorOpInBlock(*conflictingWritingOp))
1045  // conflictingWritingOp happens outside of the block. No
1046  // conflict.
1047  continue;
1048  }
1049 
1050  // No conflict if the conflicting write and the last write are the same
1051  // use.
1052  if (getAliasingOpResult(*uConflictingWrite) == lastWrite)
1053  continue;
1054 
1055  // Special rules for matching ExtractSliceOp/InsertSliceOp pairs. If
1056  // uRead is an InsertSliceOp...
1057  if (auto insertSliceOp = dyn_cast<InsertSliceOp>(readingOp)) {
1058  // As an example, consider the following IR.
1059  //
1060  // %0 = tensor.extract_slice %t[%a, %b][%c, %d][1, 1] {inplace= [true] }
1061  // %1 = linalg.fill %cst, %0 {inplace= [true] }
1062  // %2 = tensor.insert_slice %1 into %t[%a, %b][%c, %d][1, 1]
1063  // {inplace= [true] }
1064 
1065  // TODO: Use insertSliceOp.getDestOpOperand etc. when available.
1066  if (uRead == &insertSliceOp->getOpOperand(1) /*dest*/ &&
1067  hasMatchingExtractSliceOp(uConflictingWrite->get(), insertSliceOp))
1068  // Case 1: The main insight is that InsertSliceOp reads only part of
1069  // the destination tensor. The overwritten area is not read. If
1070  // uConflictingWrite writes into exactly the memory location that is
1071  // being read by uRead, this is not a conflict.
1072  //
1073  // In the above example:
1074  // uRead = OpOperand 1 (%t) of tensor.insert_slice
1075  // uConflictingWrite = OpOperand 1 (%0) of linalg.fill
1076  //
1077  // The read of %t does not conflict with the write of the FillOp
1078  // (same aliases!) because the area that the FillOp operates on is
1079  // exactly the one that is *not* read via %t.
1080  continue;
1081 
1082  if (uRead == &insertSliceOp->getOpOperand(0) /*source*/ &&
1083  uConflictingWrite == &insertSliceOp->getOpOperand(1) /*dest*/ &&
1084  hasMatchingExtractSliceOp(uRead->get(), insertSliceOp))
1085  // Case 2: The read of the source tensor and the write to the dest
1086  // tensor via an InsertSliceOp is not a conflict if the read is
1087  // reading exactly that part of an equivalent tensor that the
1088  // InsertSliceOp is writing.
1089  //
1090  // In the above example:
1091  // uRead = OpOperand 0 (%1) of tensor.insert_slice
1092  // uConflictingWrite = OpOperand 1 (%t) of tensor.insert_slice
1093  continue;
1094  }
1095 
1096  // All requirements are met. Conflict found!
1097  LDBG("CONFLICT CONFIRMED!\n\n");
1098  return true;
1099  }
1100  }
1101 
1102  LDBG("NOT A CONFLICT!\n\n");
1103  return false;
1104 }

1106 /// Return true if bufferizing result inplace would create a conflict. A read R
1107 /// and a write W of the same alias set is a conflict if inplace bufferization
1108 /// of W changes the value read by R to a value different from the one that
1109 /// would be expected by tracing back R's origin through SSA use-def chains.
1110 /// A conflict can only be introduced by a new alias and/or an inplace
1111 /// bufferization decision.
1112 ///
1113 /// Example:
1114 /// %0 = tensor.extract_slice %t[...][...][1, 1] {inplace?}
1115 /// %1 = vector.transfer_write %v1, %t {inplace} : vector<5xf32>, tensor<?xf32>
1116 /// %e = tensor.extract_slice %1
1117 /// %2 = vector.transfer_write %v2, %0 {inplace} : vector<6xf32>, tensor<?xf32>
1118 /// %3 = vector.transfer_read %e, %cst : tensor<?xf32>, vector<7xf32>
1119 ///
1120 /// In the above example, the two TransferWriteOps have already been decided to
1121 /// bufferize inplace. Bufferizing the ExtractSliceOp inplace would create a
1122 /// conflict because:
1123 /// * According to SSA use-def chains, we expect to read the result of %1.
1124 /// * However, adding an alias {%0, %t} would mean that the second

1485 /// Return the result buffer (memref) for a given OpResult (tensor). Allocate
1486 /// a new buffer and copy over data from the existing buffer if out-of-place
1487 /// bufferization is necessary.
1488 static Value getResultBuffer(OpBuilder &b, OpResult result,
1489  const BlockAndValueMapping &bvm,
1490  BufferizationAliasInfo &aliasInfo,
1491  bool skipCopy = false) {

1493  Operation *op = result.getOwner();

1495  assert(!aliasingOperands.empty() && "could not get aliasing OpOperand");

1498  assert(operandBuffer && "operand buffer not found");
1499  // Make sure that all OpOperands are the same buffer. If this is not the case,
1500  // we would have to materialize a memref value.
1501  if (!llvm::all_of(aliasingOperands, [&](OpOperand *o) {
1502  return lookup(bvm, o->get()) == operandBuffer;
1503  })) {
1504  op->emitError("result buffer is ambiguous");
1505  return Value();

1507 
1508  // If bufferizing out-of-place, allocate a new buffer.
1509  bool needCopy =
1510  getInPlace(result) != InPlaceSpec::True && !isa<scf::IfOp>(op);
1511  if (needCopy) {
1512  // Ops such as scf::IfOp can currently not bufferize out-of-place.
1513  assert(
1514  aliasingOperands.size() == 1 &&
1515  "ops with multiple aliasing OpOperands cannot bufferize out-of-place");
1516  Location loc = op->getLoc();

1518  Value resultBuffer =
1519  createNewAllocDeallocPairForShapedValue(b, loc, operand, aliasInfo);
1520  // Do not copy the result of an InitTensorOp.
1521  if (isInitTensorOp(operand))
1522  skipCopy = true;
1523  // Do not copy if the copied data is never read.
1524  if (!isValueRead(result))
1525  skipCopy = true;
1526  if (!skipCopy) {
1527  // Set insertion point now that potential alloc/dealloc are introduced.
1528  b.setInsertionPoint(op);
1529  b.create<CopyOp>(loc, operandBuffer, resultBuffer);
1530  }
1531  return resultBuffer;
1532  }

1535  return operandBuffer;

1614 /// In a first approximation, all the function arguments of a FuncOp are marked
1615 /// inplaceable. For now, it is the responsibility of the `callOp` bufferization
1616 /// to allow FuncOp that are inplaceable to write inPlace.
1617 static LogicalResult
1618 bufferize(OpBuilder &b, CallOpInterface callOp, BlockAndValueMapping &bvm,
1619  BufferizationAliasInfo &aliasInfo,
1620  DenseMap<FuncOp, FunctionType> &bufferizedFunctionTypes) {

1622  assert(isa<CallOp>(callOp.getOperation()) && funcOp &&
1623  "expected Callop to a FuncOp");
1624

1626  if (!llvm::any_of(funcOp.getType().getInputs(), isaTensor) &&
1627  !llvm::any_of(funcOp.getType().getResults(), isaTensor))

1631  OpBuilder::InsertionGuard g(b);

1635  // - if the callee is bodiless / external, we cannot inspect it and we
1636  // cannot assume anything. We can just assert that it does not return a
1637  // tensor as this would have to bufferize to "return a memref", whose
1638  // semantics is ill-defined.
1639  // - if the callee has a body, we perform inter-procedural equivalence
1640  // analysis. When successful, a result folds onto an operand. When
1641  // unsuccessful, additional work is needed to either:
1642  // * hoist a result into an inplaceable operand or
1643  // * devise a better representation to truly return a buffer.

1645  SmallVector<Value> hoistedArguments;

1650  } else {

1652  assert(returnOp && "expected func with single return op");

1655  for (OpOperand &returnOperand : returnOp->getOpOperands()) {

1663  Value returnVal = returnOperand.get();

1671  aliasInfo.insertNewBufferEquivalence(oldRes, buffer);
1672  map(bvm, oldRes, buffer);

1675  // This TensorLoadOp must fold/DCE away or bufferization should be
1676  // considered failed.
1677  Value tensorLoad =
1678  b.create<memref::TensorLoadOp>(callOp.getLoc(), buffer);
1679  oldRes.replaceAllUsesWith(tensorLoad);
1680  // Add new op equivalence info.
1681  aliasInfo.insertNewBufferEquivalence(tensorLoad, buffer);
1682  map(bvm, tensorLoad, buffer);
1683  continue;
1684  }

1687  if (Operation *allocOp = getEquivalentAlloc(returnVal, aliasInfo)) {
1688  hoistedArguments.push_back(allocOp->getResult(0));
1689  continue;
1690  }

1693  // supported. For instance, if hoisting across function boundary has
1694  // failed, it may be due to e.g. data-dependent sizes. In such a case, we
1695  // would we need a better type than memref.
1696  resultTypes.push_back(returnType);
1697 
1698  int64_t returnIdx = returnOperand.getOperandNumber();

1700  << "buffer result #" << returnIdx << " not produced by an alloc\n";

1702  }

1705  SmallVector<Type> argumentTypes{callOp->getOperandTypes()};

1707  llvm::append_range(argumentTypes, hoistedArgs.getTypes());

1709  // available.
1710  FunctionType bufferizedFuncType = getOrCreateBufferizedFunctionType(
1711  funcOp, argumentTypes, resultTypes, bufferizedFunctionTypes);
1712

1714  SmallVector<Value> newOperands;
1715  newOperands.reserve(callOp->getNumOperands());
1716  for (OpOperand &opOperand : callOp->getOpOperands()) {
1717  Value tensorOperand = opOperand.get();

1719  if (!tensorOperand.getType().isa<TensorType>()) {
1720  newOperands.push_back(tensorOperand);
1721  continue;
1722  }

1725  int64_t idx = opOperand.getOperandNumber();

1727  assert(buffer && "expected bufferized value");

1730  auto memRefType = bufferizedFuncType.getInput(idx);

1732  // conservatively turn tensors into more dynamic memref than necessary.

1734  // that will either canonicalize away or fail compilation until we can do
1735  // something better.

1739  // Add new op equivalence info.
1740  aliasInfo.insertNewBufferEquivalence(castBuffer, buffer);
1741  map(bvm, tensorOperand, castBuffer);

2583 /// the necessary work to clone operands so as to make them inplaceable.

2586 /// candidate for hoisting and creating a new inplace operand at all caller

2727 /// retrieve the called FuncOp from any CallOpInterface.
2728 static LogicalResult
2729 getFuncOpsOrderedByCalls(ModuleOp moduleOp,
2730  SmallVectorImpl<FuncOp> &orderedFuncOps,
2731  DenseMap<FuncOp, DenseSet<Operation *>> &callerMap) {

2733  // symbols of all nested CallOpInterfaceOp).

2735  // For each FuncOp, the number of CallOpInterface it contains.
2736  DenseMap<FuncOp, unsigned> numberCallOpsContainedInFuncOp;

2743  "without a unique ReturnOp";

2765  // functions remaining in the callCounter map and add them to the worklist.
2766  while (!numberCallOpsContainedInFuncOp.empty()) {

2768  [](auto entry) { return entry.getSecond() == 0; });

2810 /// This is a purely mechanical process that may later become part of a
2811 /// separate pass with its own layout assignment heuristic.
2812 static void layoutPostProcessing(ModuleOp moduleOp) {
2813  SmallVector<FuncOp> orderedFuncOps;
2814  DenseMap<FuncOp, DenseSet<Operation *>> callerMap;
2815  auto res = getFuncOpsOrderedByCalls(moduleOp, orderedFuncOps, callerMap);
2816  (void)res;
2817  assert(succeeded(res) && "unexpected getFuncOpsOrderedByCalls failure");

2819  for (FuncOp funcOp : orderedFuncOps) {
2820  DenseMap<Operation *, SmallVector<Value>> operandsPerCaller;
2821  foreachCaller(callerMap, funcOp, [&](Operation *caller) {
2822  operandsPerCaller.try_emplace(caller, SmallVector<Value>());
2823  });
2824 
2825  SmallVector<Type> argumentTypes;
2826  // Iterate on each function argument and check it it was marked with a
2827  // desired layout.
2828  for (auto it : llvm::enumerate(funcOp.getType().getInputs())) {
2829  int argNumber = it.index();
2830  Type inputType = it.value();
2831  auto memrefType = inputType.dyn_cast<MemRefType>();
2832  auto layoutAttr = funcOp.getArgAttrOfType<AffineMapAttr>(
2833  argNumber, LinalgDialect::kBufferLayoutAttrName);
2834  AffineMap desiredLayoutMap =
2835  layoutAttr ? layoutAttr.getValue() : AffineMap();
2836  AffineMap currentLayoutMap =
2837  memrefType ? getStridedLinearLayoutMap(memrefType) : AffineMap();
2838  if (!memrefType || !layoutAttr || desiredLayoutMap == currentLayoutMap) {
2839  argumentTypes.push_back(inputType);
2840  foreachCaller(callerMap, funcOp, [&](Operation *caller) {
2841  operandsPerCaller.find(caller)->getSecond().push_back(
2842  caller->getOperand(argNumber));
2843  });
2844  continue;
2845  }
2846 
2847  // Compute the buffer type with desired layout and add to input argument
2848  // types.
2849  MemRefType desiredMemrefType = MemRefType::get(
2850  memrefType.getShape(), memrefType.getElementType(), desiredLayoutMap);
2851  argumentTypes.push_back(desiredMemrefType);
2852 
2853  // If funcOp's body is not empty, change the bbArg type and propagate.
2854  if (!funcOp.body().empty()) {
2855  BlockArgument bbArg = funcOp.getArgument(argNumber);
2856  bbArg.setType(desiredMemrefType);
2857  OpBuilder b(bbArg.getContext());
2858  b.setInsertionPointToStart(bbArg.getOwner());
2859  // Cast back to the original memrefType and let it canonicalize.
2860  Value cast =
2861  b.create<memref::CastOp>(funcOp.getLoc(), memrefType, bbArg);
2862  bbArg.replaceAllUsesExcept(cast, cast.getDefiningOp());
2863  }
2864 
2865  // Cast to desired buffer type on all callers to `funcOp`.
2866  // TODO: on the callee side, this may even have to trigger a copy to
2867  // change the layout. For now let the memref::CastOp fail to verify in
2868  // such cases.

2870  OpBuilder b(caller);
2871  Value newOperand = b.create<memref::CastOp>(
2872  funcOp.getLoc(), desiredMemrefType, caller->getOperand(argNumber));
2873  operandsPerCaller.find(caller)->getSecond().push_back(newOperand);
2874  };
2875  foreachCaller(callerMap, funcOp, castArg);
2876  }
2877 
2878  // Set operands with cast buffer on all callers to `funcOp`.
2879  foreachCaller(callerMap, funcOp, [&](Operation *caller) {
2880  caller->setOperands(operandsPerCaller.lookup(caller));
2881  });

2884  auto newFuncType = FunctionType::get(moduleOp.getContext(), argumentTypes,
2885  funcOp.getType().getResults());

2887  }

2890 /// Try to eliminate InitTensorOps inside funcOp. An InitTensorOp can be
2891 /// eliminated if it is eventually inserted into another tensor (and some other
2892 /// conditions are met).
2893 ///
2894 /// E.g.:
2895 /// %0 = linalg.init_tensor
2896 /// %1 = linalg.fill(%cst, %0) {inplace = [true]}
2897 /// %2 = tensor.insert_slice %1 into %t[10][20][1]
2898 ///
2899 /// InitTensorOp elimination will try to fill %t inplace instead of filling a
2900 /// new allocation %0 and inserting it into %t. This is done by replacing the
2901 /// InitTensorOp with:
2902 ///
2903 /// %0 = tensor.extract_slice %t[10][20][1]
2904 ///
2905 /// The analysis looks for matching ExtractSliceOp/InsertSliceOp pairs and lets
2906 /// those bufferize inplace in the absence of other conflicts.
2907 ///
2908 /// Starting from an InsertSliceOp, an InitTensorOp at the end of the insert
2909 /// source's reverse use-def chain is eliminated if:
2910 /// * The InsertSliceOp was decided to bufferize inplace.
2911 /// * On the reverse use-def chain path from the InsertSliceOp to the
2912 /// InitTensorOp, all ops were decided to bufferize inplace and the buffer
2913 /// relation is "equivalent" (TODO: can be relaxed if needed).
2914 /// * The reverse use-def chain has exactly one end, which is the InitTensorOp.
2915 ///
2916 /// Note that the newly inserted ExtractSliceOp may have to bufferize
2917 /// out-of-place due to RaW conflicts.
2918 static LogicalResult runInitTensorElimination(FuncOp funcOp,
2919  BufferizationAliasInfo &aliasInfo,
2920  DominanceInfo &domInfo) {
2921  OpBuilder b(funcOp->getContext());
2922 
2923  WalkResult status = funcOp->walk([&](tensor::InsertSliceOp insertOp) {
2924  // Only inplace bufferized InsertSliceOps are eligible.
2925  if (getInPlace(insertOp->getOpResult(0)) != InPlaceSpec::True)
2926  return WalkResult::skip();
2927 
2928  SetVector<Value> maybeInitTensor =
2929  findValueInReverseUseDefChain(insertOp.source(), [](Value val) {
2930  // Continue traversal until this function returns true.
2931  OpResult opResult = val.dyn_cast<OpResult>();
2932  if (!opResult)
2933  return true;
2934  if (getInPlace(opResult) != InPlaceSpec::True)
2935  return true;
2936  // Only equivalent tensors are supported at the moment. E.g., when
2937  // taking a tensor.extract_slice of an init_tensor, we can currently
2938  // not eliminate the init_tensor.
2939  SmallVector<OpOperand *> opOperands = getAliasingOpOperand(opResult);
2940  if (!llvm::all_of(opOperands, [](OpOperand *operand) {
2941  return bufferRelation(*operand) == BufferRelation::Equivalent;
2942  }))
2943  return true;
2944  return false;
2945  });
2946  // Replace only if the InsertSliceOp source originates from exactly one
2947  // InitTensorOp.
2948  if (maybeInitTensor.size() != 1 ||
2949  !maybeInitTensor.front().getDefiningOp<InitTensorOp>())
2950  return WalkResult::skip();
2951  Value initTensor = maybeInitTensor.front();
2952 
2953  b.setInsertionPoint(initTensor.getDefiningOp());
2954  auto extractOp = b.create<tensor::ExtractSliceOp>(
2955  initTensor.getLoc(), insertOp.dest(), insertOp.getMixedOffsets(),
2956  insertOp.getMixedSizes(), insertOp.getMixedStrides());
2957  // Uses of the InitTensorOp are replaced here, but the op is not deleted.
2958  // InitTensorOps without uses are ignored by the bufferization.
2959  initTensor.replaceAllUsesWith(extractOp.result());
2960  aliasInfo.createAliasInfoEntry(extractOp.result());
2961 
2962  // Run analysis on the ExtractSliceOp.
2963  if (failed(bufferizableInPlaceAnalysis(extractOp, aliasInfo, domInfo)))
2964  return WalkResult::interrupt();
2965 
2966  // Advance to the next operation.
2967  return WalkResult::advance();
2968  });
2969 
2970  return failure(status.wasInterrupted());
2971 }

2974  ModuleOp moduleOp = getOperation();

2976 
2977  SmallVector<FuncOp> orderedFuncOps;

2979  DenseMap<FuncOp, FunctionType> bufferizedFunctionTypes;

2985  BufferizationAliasInfo aliasInfo(moduleOp);

2987  // can be fully bufferized inplace by guaranteeing the CallOp is bufferized
2988  // inplace. Therefore, we just bufferize funcOp as if none of its results were
2989  // inplaceable, detect which operands are cloned internally and decide what to
2990  // do at call sites.
2991  for (FuncOp funcOp : orderedFuncOps) {

2993  if (funcOp.body().empty())
2994  continue;

2997  // =========================

2999  // us force inplace arguments at function boundaries.
3000  // TODO: do not rely on this behavior.
3001  if (callerMap.find(funcOp) != callerMap.end())
3002  for (BlockArgument bbArg : funcOp.getArguments())
3003  if (bbArg.getType().isa<TensorType>())
3004  setInPlaceFuncArgument(bbArg);
3005

3007  if (failed(inPlaceAnalysisFuncOpBody(funcOp, aliasInfo, domInfo))) {
3008  signalPassFailure();

3010  }
3011

3013  // bufferization phase.
3014  if (failed(runInitTensorElimination(funcOp, aliasInfo, domInfo))) {

3029  }

3031  if (testAnalysisOnly)

3039  signalPassFailure();

3044  return t.isa<MemRefType, UnrankedMemRefType>();
