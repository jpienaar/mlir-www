28 /// Converts `atomic_rmw` that cannot be lowered to a simple atomic op with
29 /// AtomicRMWOpLowering pattern, e.g. with "minf" or "maxf" attributes, to
30 /// `generic_atomic_rmw` with the expanded code.
31 ///
32 /// %x = atomic_rmw "maxf" %fval, %F[%i] : (f32, memref<10xf32>) -> f32
33 ///
34 /// will be lowered to
35 ///
36 /// %x = std.generic_atomic_rmw %F[%i] : memref<10xf32> {
37 /// ^bb0(%current: f32):
38 /// %cmp = arith.cmpf "ogt", %current, %fval : f32
39 /// %new_value = select %cmp, %current, %fval : f32
40 /// atomic_yield %new_value : f32
41 /// }
42 struct AtomicRMWOpConverter : public OpRewritePattern<AtomicRMWOp> {
43 public:
44  using OpRewritePattern::OpRewritePattern;
45 
46  LogicalResult matchAndRewrite(AtomicRMWOp op,
47  PatternRewriter &rewriter) const final {
48  arith::CmpFPredicate predicate;
49  switch (op.kind()) {
50  case AtomicRMWKind::maxf:
51  predicate = arith::CmpFPredicate::OGT;
52  break;
53  case AtomicRMWKind::minf:
54  predicate = arith::CmpFPredicate::OLT;
55  break;
56  default:
57  return failure();
58  }
59 
60  auto loc = op.getLoc();
61  auto genericOp =
62  rewriter.create<GenericAtomicRMWOp>(loc, op.memref(), op.indices());
63  OpBuilder bodyBuilder =
64  OpBuilder::atBlockEnd(genericOp.getBody(), rewriter.getListener());
65 
66  Value lhs = genericOp.getCurrentValue();
67  Value rhs = op.value();
68  Value cmp = bodyBuilder.create<arith::CmpFOp>(loc, predicate, lhs, rhs);
69  Value select = bodyBuilder.create<SelectOp>(loc, cmp, lhs, rhs);
70  bodyBuilder.create<AtomicYieldOp>(loc, select);
71 
72  rewriter.replaceOp(op, genericOp.getResult());
73  return success();
74  }
75 };

77 /// Converts `memref.reshape` that has a target shape of a statically-known
78 /// size to `memref.reinterpret_cast`.
79 struct MemRefReshapeOpConverter : public OpRewritePattern<memref::ReshapeOp> {
80 public:
81  using OpRewritePattern::OpRewritePattern;
82 
83  LogicalResult matchAndRewrite(memref::ReshapeOp op,
84  PatternRewriter &rewriter) const final {
85  auto shapeType = op.shape().getType().cast<MemRefType>();
86  if (!shapeType.hasStaticShape())
87  return failure();
88 
89  int64_t rank = shapeType.cast<MemRefType>().getDimSize(0);
90  SmallVector<OpFoldResult, 4> sizes, strides;
91  sizes.resize(rank);
92  strides.resize(rank);
93 
94  Location loc = op.getLoc();
95  Value stride = rewriter.create<arith::ConstantIndexOp>(loc, 1);
96  for (int i = rank - 1; i >= 0; --i) {
97  Value size;
98  // Load dynamic sizes from the shape input, use constants for static dims.
99  if (op.getType().isDynamicDim(i)) {
100  Value index = rewriter.create<arith::ConstantIndexOp>(loc, i);
101  size = rewriter.create<memref::LoadOp>(loc, op.shape(), index);
102  if (!size.getType().isa<IndexType>())
103  size = rewriter.create<arith::IndexCastOp>(loc, size,
104  rewriter.getIndexType());
105  sizes[i] = size;
106  } else {
107  sizes[i] = rewriter.getIndexAttr(op.getType().getDimSize(i));
108  size =
109  rewriter.create<arith::ConstantOp>(loc, sizes[i].get<Attribute>());
110  }
111  strides[i] = stride;
112  if (i > 0)
113  stride = rewriter.create<arith::MulIOp>(loc, stride, size);
114  }
115  rewriter.replaceOpWithNewOp<memref::ReinterpretCastOp>(
116  op, op.getType(), op.source(), /*offset=*/rewriter.getIndexAttr(0),
117  sizes, strides);
118  return success();
119  }
120 };

122 template <typename OpTy, arith::CmpFPredicate pred>
123 struct MaxMinFOpConverter : public OpRewritePattern<OpTy> {
124 public:
125  using OpRewritePattern<OpTy>::OpRewritePattern;
126 
127  LogicalResult matchAndRewrite(OpTy op,
128  PatternRewriter &rewriter) const final {
129  Value lhs = op.lhs();
130  Value rhs = op.rhs();
131 
132  Location loc = op.getLoc();
133  Value cmp = rewriter.create<arith::CmpFOp>(loc, pred, lhs, rhs);
134  Value select = rewriter.create<SelectOp>(loc, cmp, lhs, rhs);
135 
136  auto floatType = getElementTypeOrSelf(lhs.getType()).cast<FloatType>();
137  Value isNaN = rewriter.create<arith::CmpFOp>(loc, arith::CmpFPredicate::UNO,
138  lhs, rhs);
139 
140  Value nan = rewriter.create<arith::ConstantFloatOp>(
141  loc, APFloat::getQNaN(floatType.getFloatSemantics()), floatType);
142  if (VectorType vectorType = lhs.getType().dyn_cast<VectorType>())
143  nan = rewriter.create<SplatOp>(loc, vectorType, nan);
144 
145  rewriter.replaceOpWithNewOp<SelectOp>(op, isNaN, nan, select);
146  return success();
147  }
148 };

166 struct StdExpandOpsPass : public StdExpandOpsBase<StdExpandOpsPass> {
167  void runOnFunction() override {
168  MLIRContext &ctx = getContext();
169 
170  RewritePatternSet patterns(&ctx);
171  populateStdExpandOpsPatterns(patterns);
172  arith::populateArithmeticExpandOpsPatterns(patterns);
173 
174  ConversionTarget target(getContext());
175 
176  target.addLegalDialect<arith::ArithmeticDialect, memref::MemRefDialect,
177  StandardOpsDialect>();
178  target.addIllegalOp<arith::CeilDivSIOp, arith::FloorDivSIOp>();
179  target.addDynamicallyLegalOp<AtomicRMWOp>([](AtomicRMWOp op) {
180  return op.kind() != AtomicRMWKind::maxf &&
181  op.kind() != AtomicRMWKind::minf;
182  });
183  target.addDynamicallyLegalOp<memref::ReshapeOp>([](memref::ReshapeOp op) {
184  return !op.shape().getType().cast<MemRefType>().hasStaticShape();
185  });
186  // clang-format off
187  target.addIllegalOp<
188  MaxFOp,
189  MaxSIOp,
190  MaxUIOp,
191  MinFOp,
192  MinSIOp,
193  MinUIOp
194  >();
195  // clang-format on
196  if (failed(
197  applyPartialConversion(getFunction(), target, std::move(patterns))))
198  signalPassFailure();
199  }
200 };
