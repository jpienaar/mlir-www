116 /// Codegen strategy controls how a Linalg op is progressively lowered.
117 struct CodegenStrategy {
118  /// Append a pattern to add a level of tiling for Op `opName` with tiling
119  /// `options`.
120  CodegenStrategy &
121  tile(StringRef opName, linalg::LinalgTilingOptions options,
122  LinalgTransformationFilter::FilterFunction f = nullptr) {
123  transformationSequence.emplace_back(
124  std::make_unique<Tile>(opName, options, f));
125  return *this;
126  }
127  /// Conditionally append a pattern to add a level of tiling for
128  /// `LinalgOpType` with tiling `options`.
129  CodegenStrategy &
130  tileIf(bool b, StringRef opName, linalg::LinalgTilingOptions options,
131  LinalgTransformationFilter::FilterFunction f = nullptr) {
132  return b ? tile(opName, options) : *this;
133  }
134  /// Append a pattern to add a level of promotion for `LinalgOpType` with
135  /// promotion `options`.
136  CodegenStrategy &
137  promote(StringRef opName, linalg::LinalgPromotionOptions options,
138  LinalgTransformationFilter::FilterFunction f = nullptr) {
139  transformationSequence.emplace_back(
140  std::make_unique<Promote>(opName, options, f));
141  return *this;
142  }
143  /// Conditionally append a pattern to add a level of promotion for
144  /// `LinalgOpType` with promotion `options`.
145  CodegenStrategy &
146  promoteIf(bool b, StringRef opName, linalg::LinalgPromotionOptions options,
147  LinalgTransformationFilter::FilterFunction f = nullptr) {
148  return b ? promote(opName, options, f) : *this;
149  return *this;
150  }
151  /// Append a pattern to generalize named operations.
152  CodegenStrategy &
153  generalize(StringRef opName,
154  LinalgTransformationFilter::FilterFunction f = nullptr) {
155  transformationSequence.emplace_back(
156  std::make_unique<Generalize>(opName, f));
157  return *this;
158  }
159  /// Conditionally append a pattern to generalize named operations.
160  CodegenStrategy &
161  generalizeIf(bool b, StringRef opName,
162  LinalgTransformationFilter::FilterFunction f = nullptr) {
163  return b ? generalize(opName, f) : *this;
164  return *this;
165  }
166  /// Append a pattern to interchange iterators.
167  CodegenStrategy &
168  interchange(ArrayRef<int64_t> iteratorInterchange,
169  LinalgTransformationFilter::FilterFunction f = nullptr) {
170  transformationSequence.emplace_back(
171  std::make_unique<Interchange>(iteratorInterchange, f));
172  return *this;
173  }
174  /// Conditionally append a pattern to interchange iterators.
175  CodegenStrategy &
176  interchangeIf(bool b, ArrayRef<int64_t> iteratorInterchange,
177  LinalgTransformationFilter::FilterFunction f = nullptr) {
178  return b ? interchange(iteratorInterchange, f) : *this;
179  return *this;
180  }
181  /// Append a pattern to rewrite `LinalgOpType` as a vector operation.
182  CodegenStrategy &
183  vectorize(StringRef opName,
184  LinalgTransformationFilter::FilterFunction f = nullptr) {
185  assert(!opName.empty() && "expected an op name");
186  transformationSequence.emplace_back(std::make_unique<Vectorize>(
187  opName, linalg::LinalgVectorizationOptions(), f));
188  return *this;
189  }
190  /// Conditionally append a pattern to rewrite `LinalgOpType` as a vector
191  /// operation.
192  CodegenStrategy &
193  vectorizeIf(bool b, StringRef opName,
194  LinalgTransformationFilter::FilterFunction f = nullptr) {
195  return b ? vectorize(opName, f) : *this;
196  return *this;
197  }
198  /// Configure the post staged-patterns late vector transformations.
199  CodegenStrategy &
200  setVectorTransformsOptions(vector::VectorTransformsOptions options) {
201  vectorTransformOptions = options;
202  return *this;
203  }
204  /// Configure the post staged-patterns late vector.transfer to scf
205  /// conversion.
206  CodegenStrategy &
207  setVectorTransferToSCFOptions(VectorTransferToSCFOptions options) {
208  vectorToSCFOptions = options;
209  return *this;
210  }
211  ///
212  /// Configure the application of late transformations.
