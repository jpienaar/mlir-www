46 // A helper RAII class to emit nested namespaces for this op.
47 class NamespaceEmitter {
48 public:
49  NamespaceEmitter(raw_ostream &os, const Dialect &dialect) : os(os) {
50  if (!dialect)
51  return;
52  emitNamespaceStarts(os, dialect.getCppNamespace());
53  }
54  NamespaceEmitter(raw_ostream &os, StringRef cppNamespace) : os(os) {
55  emitNamespaceStarts(os, cppNamespace);
56  }
57 
58  ~NamespaceEmitter() {
59  for (StringRef ns : llvm::reverse(namespaces))
60  os << "} // namespace " << ns << "\n";
61  }
62 
63 private:
64  void emitNamespaceStarts(raw_ostream &os, StringRef cppNamespace) {
65  llvm::SplitString(cppNamespace, namespaces, "::");
66  for (StringRef ns : namespaces)
67  os << "namespace " << ns << " {\n";
68  }
69  raw_ostream &os;
70  SmallVector<StringRef, 2> namespaces;
71 };

73 /// This class deduplicates shared operation verification code by emitting
74 /// static functions alongside the op definitions. These methods are local to
75 /// the definition file, and are invoked within the operation verify methods.
76 /// An example is shown below:
77 ///
78 /// static LogicalResult localVerify(...)
79 ///
80 /// LogicalResult OpA::verify(...) {
81 /// if (failed(localVerify(...)))
82 /// return failure();
83 /// ...
84 /// }
85 ///
86 /// LogicalResult OpB::verify(...) {
87 /// if (failed(localVerify(...)))
88 /// return failure();
89 /// ...
90 /// }
91 ///
92 class StaticVerifierFunctionEmitter {
93 public:
94  StaticVerifierFunctionEmitter(const llvm::RecordKeeper &records,
95  raw_ostream &os);
96 
97  /// Emit the static verifier functions for `llvm::Record`s. The
98  /// `signatureFormat` describes the required arguments and it must have a
99  /// placeholder for function name.
100  /// Example,
101  /// const char *typeVerifierSignature =
102  /// "static ::mlir::LogicalResult {0}(::mlir::Operation *op, ::mlir::Type"
103  /// " type, ::llvm::StringRef valueKind, unsigned valueGroupStartIndex)";
104  ///
105  /// `errorHandlerFormat` describes the error message to return. It may have a
106  /// placeholder for the summary of Constraint and bring more information for
107  /// the error message.
108  /// Example,
109  /// const char *typeVerifierErrorHandler =
110  /// " op->emitOpError(valueKind) << \" #\" << valueGroupStartIndex << "
111  /// "\" must be {0}, but got \" << type";
112  ///
113  /// `typeArgName` is used to identify the argument that needs to check its
114  /// type. The constraint template will replace `$_self` with it.
115  void emitFunctionsFor(StringRef signatureFormat, StringRef errorHandlerFormat,
116  StringRef typeArgName, ArrayRef<llvm::Record *> opDefs,
117  bool emitDecl);
118 
119  /// Get the name of the local function used for the given type constraint.
120  /// These functions are used for operand and result constraints and have the
121  /// form:
122  /// LogicalResult(Operation *op, Type type, StringRef valueKind,
123  /// unsigned valueGroupStartIndex);
124  StringRef getTypeConstraintFn(const Constraint &constraint) const;
125 
126 private:
127  /// Returns a unique name to use when generating local methods.
128  static std::string getUniqueName(const llvm::RecordKeeper &records);
129 
130  /// Emit local methods for the type constraints used within the provided op
131  /// definitions.
132  void emitTypeConstraintMethods(StringRef signatureFormat,
133  StringRef errorHandlerFormat,

142  std::string uniqueOutputLabel;

144  /// A set of functions implementing type constraints, used for operand and
145  /// result verification.
146  llvm::DenseMap<const void *, std::string> localTypeConstraints;
