25 /// The BufferizationAliasInfo class maintains a list of buffer aliases and
26 /// equivalence classes to support bufferization.
27 /// ExtractSliceOps have special behavior, they act as a level of indirection
28 /// for bufferization. They don't create reads or writes themselves and analysis
29 /// needs to look through their uses.
30 /// ExtractSliceOp + InsertSliceOp have special joint behavior: they may
31 /// bufferize to the same buffer (i.e. subview), which is what introduces the
32 /// need for bufferization classes.
33 /// Some of these functionalities could be refactored in a Bufferizer class that
34 /// uses BufferizationAliasInfo.
35 class BufferizationAliasInfo {
36 public:
37  /// Specify fine-grain relationship between buffers to enable more analysis.
38  enum class BufferRelation {
39  None,
40  // TODO: ResultContainsOperand,
41  // TODO: OperandContainsResult,
42  Equivalent
43  };

48  /// beginning the alias and equivalence sets only contain `v` itself.

119 private:
120  /// llvm::EquivalenceClasses wants comparable elements because it uses
121  /// std::set as the underlying impl.
122  /// ValueWrapper wraps Value and uses pointer comparison on the defining op.
123  /// This is a poor man's comparison but it's not like UnionFind needs ordering
124  /// anyway ..
125  struct ValueWrapper {
126  ValueWrapper(Value val) : v(val) {}
127  operator Value() const { return v; }
128  bool operator<(const ValueWrapper &wrap) const {
129  return v.getImpl() < wrap.v.getImpl();
130  }
131  bool operator==(const ValueWrapper &wrap) const { return v == wrap.v; }
132  Value v;
133  };

135  using EquivalenceClassRangeType = llvm::iterator_range<
136  llvm::EquivalenceClasses<ValueWrapper>::member_iterator>;
137  /// Check that aliasInfo for `v` exists and return a reference to it.
138  EquivalenceClassRangeType getAliases(Value v) const;
139 
140  /// Return true if the (ExtractSliceOp, InsertSliceOp) pair match (i.e.
141  /// equivalent operand / result and same offset/sizes/strides specification).
142  ///
143  /// This is one particular type of relationship between ops on tensors that
144  /// reduce to an equivalence on buffers. This should be generalized and
145  /// exposed as interfaces on the proper types.
146  bool areEquivalentExtractSliceOps(tensor::ExtractSliceOp st,

149  /// Given sets of uses and writes, return true if there is a RaW conflict
150  /// under the assumption that all given reads/writes alias the same buffer and
151  /// that all given writes bufferize inplace.
152  bool hasReadAfterWriteInterference(const DenseSet<OpOperand *> &usesRead,
153  const DenseSet<OpOperand *> &usesWrite,
154  const DominanceInfo &domInfo) const;
155 
156  /// Set of tensors that are known to bufferize to writable memory.
157  llvm::DenseSet<Value> bufferizeToWritableMemory;

159  /// Auxiliary structure to store all the values a given value aliases with.

162  llvm::EquivalenceClasses<ValueWrapper> aliasInfo;
163 
164  /// Auxiliary structure to store all the equivalent buffer classes.
165  llvm::EquivalenceClasses<ValueWrapper> equivalentInfo;
166 };
